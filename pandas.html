  • Days when it was between 50 and 60
  • Days when it was anything else
def grouper(df, idx, col):
    if 80 <= df[col].loc[idx] <= 90:
        return '80-90'
    elif 50 <= df[col].loc[idx] <= 60:
        return '50-60'
    else:
        return 'others'


g = df.groupby(lambda x: grouper(df, x, 'temperature'))


for key, d in g:
